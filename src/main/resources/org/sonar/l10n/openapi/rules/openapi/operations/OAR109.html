500 Internal Server Error response codes should not be defined

Noncompliant Code Example (OpenAPI 2)

    swagger: "2.0"
    info:
      version: "1.0.0"
      title: "Swagger Petstore"
    host: "example.com"
    basePath: "/"
    schemes:
      - "https"
    consumes:
      - "application/json"
    produces:
      - "application/json"
    paths:
      /status:
        delete:
          summary: "Get API status"
          responses:
            "400":
              description: "error"
      /another:
        delete:
          summary: "Get another API status"
          responses:
            "400":
              description: "error"
      /users:
        delete: 
          summary: "Get all users"
          responses: # Noncompliant {{OAR109: 500 Internal Server Error response codes should not be defined}}
            "400":
              description: "Error"
            "500":
              description: "Internal Server Error"
