The defined media types must conform to the RFC6838 and RFC7231. This ensures that a tool can rely on the declared media type to generate appropriate code to handle the parsing of output.

Noncompliant Code Example (OpenAPI 2)

JSON

{
    "swagger": "2.0", 
    "info": {
        "version": "1.0.0", 
        "title": "Swagger Petstore"
    }, 
    "produces": [
        "application/json", 
        "text/*"
    ], 
    "consumes": [
        "application/json", 
        "text/*"
    ], 
    "paths": {
        "/pets": null, 
        "get": {
            "produces": [
                "application", 
                "text/plain"
            ], 
            "consumes": [
                "application", 
                "text/plain"
            ], 
            "responses": null, 
            "200": {
                "description": "some operation"
            }
        }
    }
}

YAML

swagger: "2.0"
info:
    version: 1.0.0
    title: Swagger Petstore
produces:
    - application/json
    - text/*
consumes:
    - application/json
    - text/*
paths:
    /pets:
    get:
        produces:
        - application
        - text/plain
        consumes:
        - application
        - text/plain
        responses:
        '200':
            description: some operation
