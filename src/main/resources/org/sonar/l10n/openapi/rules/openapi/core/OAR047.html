Each tag should be documented with a short statement explaining its intent.

Tags must start with capital letter

Noncompliant Code Example (OpenAPI 2)

JSON

{
    "swagger": "2.0", 
    "info": {
        "version": "1.0.0", 
        "title": "Swagger Petstore"
    }, 
    "tags": [
        {
            "name": "unused-tag"
        }, 
        {
            "name": "some-tag"
        }
    ], 
    "paths": {
        "/pets": {
            "get": {
                "tags": [
                    "some-tag"
                ], 
                "responses": {
                    "200": {
                        "description": "OK"
                    }
                }
            }
        }
    }
}

YAML

swagger: "2.0"
info:
  version: 1.0.0
  title: Swagger Petstore
tags:
- name: unused-tag
- name: some-tag
paths:
  /pets:
    get:
      tags:
      - some-tag
      responses:
        200:
          description: OK

Compliant Solution (OpenAPI 2)

JSON

{
    "swaggeer": "2.0", 
    "info": {
        "version": "1.0.0", 
        "title": "Swagger Petstore"
    }, 
    "tags": [
        {
            "name": "Some-tag", 
            "description": "a tag referenced in the operations"
        }, 
        {
            "name": "Other-tag", 
            "description": "another tag referenced in the operations"
        }
    ], 
    "paths": {
        "/pets": {
            "get": {
                "tags": [
                    "Some-tag", 
                    "Other-tag"
                ], 
                "responses": {
                    "200": {
                        "description": "OK"
                    }
                }
            }
        }
    }
}

YAML

swaggeer: "2.0"
info:
  version: 1.0.0
  title: Swagger Petstore
tags:
- name: Some-tag
  description: a tag referenced in the operations
- name: Other-tag
  description: another tag referenced in the operations
paths:
  /pets:
    get:
      tags:
      - Some-tag
      - Other-tag
      responses:
        200:
          description: OK
