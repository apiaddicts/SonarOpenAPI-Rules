In "get" operations, the $select parameter, which allows to indicate the attributes that will be returned in the response, must be defined as a non-required parameter.

Noncompliant Code Example (OpenAPI 2)

JSON

{
    "swagger": "2.0", 
    "info": {
        "title": "Swagger Petstore", 
        "version": "1.0.0"
    }, 
    "paths": {
        "/pets": {
            "get": {
                "responses": {
                    "200": {
                        "description": "Ok"
                    }
                }
            }
        }
    }
}

YAML

swagger: "2.0"
info:
  title: Swagger Petstore
  version: 1.0.0
paths:
  /pets:
    get:
      responses:
        200:
          description: Ok

Compliant Solution (OpenAPI 2)

JSON

{
    "swagger": "2.0", 
    "info": {
        "title": "Swagger Petstore", 
        "version": "1.0.0"
    }, 
    "paths": {
        "/pets": {
            "get": {
                "parameters": [
                    {
                        "in": "query", 
                        "name": "$select", 
                        "required": false, 
                        "type": "array", 
                        "items": {
                            "type": "string"
                        }
                    }
                ], 
                "responses": {
                    "200": {
                        "description": "Ok"
                    }
                }
            }
        }
    }
}

YAML

swagger: "2.0"
info:
  title: Swagger Petstore
  version: 1.0.0
paths:
  /pets:
    get:
      parameters:
        - in: query
          name: $select
          required: false
          type: array
          items:
            type: string
      responses:
        200:
          description: Ok
