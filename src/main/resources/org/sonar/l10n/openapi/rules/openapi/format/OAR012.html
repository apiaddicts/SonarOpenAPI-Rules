The convention for naming properties in a JSON is "snake_case" (the words must be in lowercase and separated by an underscore), therefore, the path parameters, the query parameters, request body properties and response body properties must be in this convention.

Noncompliant Code Example (OpenAPI 2)

All the examples are in YAML

  swagger: "2.0"
  info:
    title: Swagger Petstore
    version: "1.0"
  paths:
    /pets/{pet-id}:
      patch:
        consumes:
          - application/json
        produces:
          - application/json
        parameters:
          - in: path
            name: pet-id
            type: integer
            required: true
          - in: query
            name: pet_Type
            type: string
          - in: body
            name: pet
            description: Pet
            schema:
              type: object
              required:
                - PET_NAME
              properties:
                PET_NAME:
                  type: string
                photoUrls:
                  type: array
                  items:
                    type: string
        responses:
          200:
            description: OK
            schema:
              type: object
              properties:
                petId:
                  type: integer
                petType:
                  type: string
                petName:
                  type: string
                photoUrls:
                  type: array
                  items:
                    type: string

  info:

      patch:
        consumes:
