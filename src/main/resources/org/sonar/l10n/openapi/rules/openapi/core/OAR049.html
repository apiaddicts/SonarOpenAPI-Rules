Responses with code 204 should not return content.

Noncompliant Code Example (OpenAPI 2)

JSON

{
    "swagger": "2.0", 
    "info": {
        "version": "1.0.0", 
        "title": "Swagger Petstore"
    }, 
    "paths": {
        "/pets/{petId}": {
            "delete": {
                "parameters": [
                    {
                        "in": "path", 
                        "name": "petId", 
                        "required": true, 
                        "type": "string", 
                        "format": "uuid"
                    }
                ], 
                "responses": {
                    "204": {
                        "description": "Deleted", 
                        "schema": {
                            "type": "object"
                        }
                    }
                }
            }
        }
    }
}

YAML

swagger: "2.0"
info:
  version: 1.0.0
  title: Swagger Petstore
paths:
  /pets/{petId}:
    delete:
      parameters:
        - in: path
          name: petId
          required: true
          type: string
          format: uuid
      responses:
        204:
          description: Deleted
          schema:
            type: object
