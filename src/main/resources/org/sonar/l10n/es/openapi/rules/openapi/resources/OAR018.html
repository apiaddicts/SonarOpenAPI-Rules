Dependiendo del verbo, no se recomiendan algunas estructuras de path. Esto puede ser un indicio de un defecto de diseño.

Estructuras permitidas

VERBO ESTRUCTURA
GET /resources
GET /resources/{id}
GET /resources/{id}/subresources
POST /resources
POST /resources/get
POST /resources/delete
POST /resources/{id}/subresources
PUT /resources/{id}
PATCH /resources/{id}
DELETE /resources/{id}

Ejemplo de código no compatible (OpenAPI 2)

JSON

{
    "swagger": "2.0", 
    "info": {
        "title": "Swagger Petstore", 
        "version": "1.0.0"
    }, 
    "paths": {
        "/customers/{customerId}": {
            "post": {
                "parameters": [
                    {
                        "name": "customerId", 
                        "in": "path", 
                        "required": true, 
                        "type": "integer"
                    }, 
                    {
                        "name": "body", 
                        "in": "body", 
                        "required": true, 
                        "schema": {
                            "type": "object"
                        }
                    }
                ], 
                "responses": {
                    "200": {
                        "description": "Ok"
                    }
                }
            }
        }
    }
}

YAML

swagger: "2.0"
info:
  title: Swagger Petstore
  version: 1.0.0
paths:
  /customers/{customerId}:
    post:
      parameters:
      - name: customerId
        in: path
        required: true
        type: integer
      - name: body
        in: body
        required: true
        schema:
          type: object
      responses:
        200:
          description: Ok
