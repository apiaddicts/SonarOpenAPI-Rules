Se deben utilizar códigos de respuesta coherentes en cada operación.

Ejemplo de código no compatible (OpenAPI 2)

JSON

{
    "swagger": "2.0", 
    "info": {
        "version": "1.0.0", 
        "title": "Swagger Petstore"
    }, 
    "paths": {
        "/pets": {
            "get": {
                "responses": {
                    "default": {
                        "description": "Default"
                    }
                }
            }, 
            "post": {
                "responses": {
                    "default": {
                        "description": "Default"
                    }
                }
            }
        }
    }
}

YAML

swagger: "2.0"
info:
  version: 1.0.0
  title: Swagger Petstore
paths:
  /pets:
    get:
      responses:
        default:
          description: Default
    post:
      responses:
        default:
          description: Default

Solución compatible (OpenAPI 2)

JSON

{
    "swagger": "2.0", 
    "info": {
        "version": "1.0.0", 
        "title": "Swagger Petstore"
    }, 
    "paths": {
        "/pets": {
            "get": {
                "responses": {
                    "206": {
                        "description": "Partial collection"
                    }, 
                    "400": {
                        "description": "Bad request"
                    }, 
                    "500": {
                        "description": "Internal server error"
                    }, 
                    "504": {
                        "description": "Gateway timeout"
                    }, 
                    "default": {
                        "description": "Default"
                    }
                }
            }, 
            "post": {
                "responses": {
                    "201": {
                        "description": "Created"
                    }, 
                    "400": {
                        "description": "Bad request"
                    }, 
                    "413": {
                        "description": "Payload too large"
                    }, 
                    "415": {
                        "description": "Unsupported media type"
                    }, 
                    "500": {
                        "description": "Internal server error"
                    }, 
                    "504": {
                        "description": "Gateway timeout"
                    }, 
                    "default": {
                        "description": "Default"
                    }
                }
            }
        }
    }
}

YAML

swagger: "2.0"
info:
  version: 1.0.0
  title: Swagger Petstore
paths:
  /pets:
    get:
      responses:

          description: Partial collection

          description: Bad request
        500:
          description: Internal server error
        504:
          description: Gateway timeout
        default:
          description: Default
    post:
      responses:
        201:
          description: Created
        400:
          description: Bad request
        413:
          description: Payload too large
        415:
          description: Unsupported media type
        500:
          description: Internal server error
        504:
          description: Gateway timeout
        default:
          description: Default
