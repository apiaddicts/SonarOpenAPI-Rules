En operaciones de consulta, el parámetro $filter, que permite indicar los filtros, se debe definir como parámetro no obligatorio.

Ejemplo de código no compatible (OpenAPI 2)

JSON

{
    "swagger": "2.0", 
    "info": {
        "title": "Swagger Petstore", 
        "version": "1.0.0"
    }, 
    "paths": {
        "/pets": {
            "get": {
                "responses": {
                    "200": {
                        "description": "Ok"
                    }
                }
            }
        }
    }
}

YAML

swagger: "2.0"
info:
  title: Swagger Petstore
  version: 1.0.0
paths:
  /pets:
    get:
      responses:
        200:
          description: Ok

Solución compatible (OpenAPI 2)

JSON

{
    "swagger": "2.0", 
    "info": {
        "title": "Swagger Petstore", 
        "version": "1.0.0"
    }, 
    "paths": {
        "/pets": {
            "get": {
                "parameters": [
                    {
                        "in": "query", 
                        "name": "$filter", 
                        "required": false, 
                        "type": "object"
                    }
                ], 
                "responses": {
                    "200": {
                        "description": "Ok"
                    }
                }
            }
        }
    }
}

YAML

swagger: "2.0"
info:
  title: Swagger Petstore
  version: 1.0.0
paths:
  /pets:
    get:
      parameters:
        - in: query
          name: $filter
          required: false
          type: object
      responses:
        200:
          description: Ok
